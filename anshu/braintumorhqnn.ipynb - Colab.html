@qml.qnode(dev)
def quantum_convolution(weights, x):
    for i in range(4):
        qml.RX(x[i % len(x)], wires=i)
    for i in range(4):
        qml.RY(weights[0][i], wires=i)
        qml.RZ(weights[1][i], wires=i)
    for i in range(3):
        qml.CNOT(wires=[i, i + 1])
    return [qml.expval(qml.PauliZ(w)) for w in range(4)]
